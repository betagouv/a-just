import { CommonModule } from '@angular/common'
import { Component, Input } from '@angular/core'

/**
 * Composant de génération des tooltips
 */

@Component({
  selector: 'aj-completion-bar',
  standalone: true,
  imports: [CommonModule],
  templateUrl: './completion-bar.component.html',
  styleUrls: ['./completion-bar.component.scss'],
})
export class CompletionBarComponent {
  /**
   * Valeur du pourcentage
   */
  @Input() value: number = 0
}

<div class="progression-bar">
  <div class="bar">
    <div
      class="level"
      [ngClass]="{
        rouge: this.value <= 50,
        jaune: this.value > 50 && this.value <= 80,
        bleu: this.value > 80 && this.value <= 99,
        vert: this.value > 99 && this.value <= 100,
      }"
      [ngStyle]="{ width: this.value + '%' }"
    ></div>
  </div>
</div>
